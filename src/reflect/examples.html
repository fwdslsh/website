Usage Tracking Workflows

Practical analytics patterns

Daily Usage Analysis

# Analyze daily usage
reflect analyze usage \
  --time-start $(date -d yesterday +%Y-%m-%dT00:00:00Z) \
  --time-end $(date -d today +%Y-%m-%dT23:59:59Z)

Confidence Scoring

# Track learning outcomes
reflect use learning_abc123

# Record successful application
reflect outcome learning_abc123 --result success

# Record learning failure
reflect outcome learning_abc123 --result failure

Knowledge Gap Analysis

Identify Missing Learnings

# Find low-confidence learnings
reflect analyze learnings --max-confidence 0.6

# Find learnings not used recently
reflect analyze usage --time-start $(date -d '30 days ago' +%Y-%m-%dT00:00:00Z)

Trend Analysis

# Analyze learning creation trends
reflect analyze learnings --time-start 2025-01-01T00:00:00Z

Automation Examples

Weekly Reports

#!/bin/bash
# weekly-report.sh
START=$(date -d '7 days ago' +%Y-%m-%dT00:00:00Z)
END=$(date +%Y-%m-%dT23:59:59Z)

reflect analyze usage --time-start "$START" --time-end "$END" \
  --output "weekly-$(date +%Y-%m-%d).md"

DataHarvester API

Programmatic Access

import { createHarvesterConfig, DataHarvester } from "@fwdslsh/reflect";

const config = createHarvesterConfig();
const harvester = new DataHarvester(config);

// Harvest all data types
const [usage, procedures, learnings, sessions] = await Promise.all([
  harvester.harvestUsage({ limit: 100 }),
  harvester.harvestProcedures(),
  harvester.harvestLearnings({ minConfidence: 0.7 }),
  harvester.harvestSessions()
]);
