Learning Capture Examples

Capturing Patterns

# During development, capture discovered patterns
inscribe learning add \
  --type pattern \
  --summary "Bun.serve() handles async differently" \
  --content-file pattern-notes.md \
  --domains bun,http \
  --confidence 0.85 \
  --session $HYPHN_SESSION_ID

Recording Problem-Solutions

inscribe learning add \
  --type problem-solution \
  --summary "Fix Bun test stdin reading" \
  --content "## Problem\nBun tests hang when reading stdin\n\n## Solution\nUse --content or --content-file flags explicitly" \
  --domains bun,testing \
  --confidence 0.9

Mistake Documentation

inscribe learning add \
  --type mistake \
  --summary "Don't use fs.readFile() in Bun.serve()" \
  --content "Bun.serve() request handlers cannot read from stdin. Always read from file or use --content flag." \
  --domains bun,http \
  --confidence 0.95

Session Tracking Examples

Feature Development Session

# Start session
inscribe session start \
  --name "implement-auth" \
  --repo /path/to/project \
  --branch feature/auth \
  --tags "auth,security"

# Work happens...

# Stop with success
inscribe session stop --status ok

Failed Session

inscribe session stop --status failed

CI/CD Integration

GitHub Actions

name: Record Learning
on:
  workflow_run:
    types:
      - completed

jobs:
  record:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Record learning
        run: |
          # Extract learning from PR description or comments
          # and record it
          inscribe learning add \
            --type pattern \
            --summary "$LEARNING_SUMMARY" \
            --content "$LEARNING_CONTENT"
