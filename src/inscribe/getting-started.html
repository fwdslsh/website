Getting Started

Installation and basic usage

Installation

# From monorepo root
npm install
# or
bun install

# Available in workspace

Learning Capture

# Add a learning
inscribe learning add \
  --type pattern \
  --summary "Use Bun native APIs" \
  --content "Prefer Bun.file() over fs.readFile()" \
  --domains bun,typescript \
  --confidence 0.9

# Read from file
inscribe learning add \
  --type problem-solution \
  --content-file solution.md \
  --domains bun

Session Management

# Start session
inscribe session start \
  --name "feature-work" \
  --repo /path/to/repo \
  --branch feature

# Stop session
inscribe session stop --status ok

Event Recording

# Record decision
inscribe add decision "Switch to TypeScript strict mode" \
  --tags architecture,typescript

# Record event
echo "Task completed" | inscribe add event --stdin
