Integration with Hyphn Tools

# Capture a learning
inscribe learning add \
  --type pattern \
  --summary "Use Bun.spawn for subprocesses" \
  --domains "bun" \
  --confidence 0.9

# Search learnings
disclose learnings --query "bun performance" --domains bun

# Record usage
reflect use learn_2025-01-15_a3f9b2c4

# Record outcome
reflect outcome learn_2025-01-15_a3f9b2c4 --result success
